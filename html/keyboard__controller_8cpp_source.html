167 #pragma GCC diagnostic push
168 #pragma GCC diagnostic ignored "-Wfloat-equal"
169  double steer = this->m_key_pressed_state[(size_t)KeyIndex::STEER_LEFT]
170  ? -1
171  : (this->m_key_pressed_state[(size_t)KeyIndex::STEER_RIGHT] ? +1 : 0);
172  double throttle = this->m_key_pressed_state[(size_t)KeyIndex::ACCELERATE]
173  ? +1
174  : (this->m_key_pressed_state[(size_t)KeyIndex::DECELERATE] ? -1 : 0);
175 
176  double steer_limit = map(std::abs(this->m_velocity), 0, MAX_THROTTLE, 1, FAST_STEER_LIMIT);
177  double angle_update = steer * delta_time * STEERING_SPEED;
178  this->m_angle = boost::algorithm::clamp(this->m_angle + angle_update, -steer_limit, +steer_limit);
179  double velocity_update = throttle * delta_time * (this->m_velocity * throttle > 0 ? ACCELERATION : BRAKING);
180  this->m_velocity = boost::algorithm::clamp(this->m_velocity + velocity_update, -MAX_THROTTLE, +MAX_THROTTLE);
181 
182  if (steer == 0 && this->m_angle != 0)
183  {
184  double sign = std::copysign(1.0, this->m_angle);
185  this->m_angle -= STEERING_GRAVITY * delta_time * sign;
186  if (std::abs(this->m_angle) < STEERING_GRAVITY * delta_time)
187  {
188  this->m_angle = 0;
189  }
190  }
191 
192  if (throttle == 0 && this->m_velocity != 0)
193  {
